\section{Giả thuyết Riemann}
\begin{frame}{\secname}
	\begin{itemize}
		\item Lịch sử nghiên cứu
		\item Hàm $\zeta$
			\begin{equation*}
				\zeta(s)=\sum_{n=1}^\infty \frac{1}{n^s} = \prod_p \frac{1}{1-p^{-s}}
			\end{equation*}
		\item Phương trình hàm $\zeta$
			\begin{equation*}
				\zeta(s) = \frac{s}{s-1}- s \int_{1}^{\infty} \frac{\{x\}}{x^{s+1}}dx
			\end{equation*}
	\end{itemize}	
\end{frame}
